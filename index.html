# Heart Disease Prediction - Team Solution

# Step 1: Import Libraries
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import confusion_matrix, accuracy_score, classification_report

# Step 2: Load Data
data = pd.read_csv('heart.csv')
print(f"Dataset loaded: {data.shape}")

# Step 3: Check for missing values
print(data.isnull().sum())

# Step 4: Data Exploration
plt.hist(data['HeartDisease'])
plt.title('Heart Disease Distribution')
plt.show()

# Visualize Chest Pain vs Heart Disease
sns.countplot(x='ChestPainType', hue='HeartDisease', data=data)
plt.title('Chest Pain Type by Heart Disease')
plt.show()

# Step 5: Preprocessing - Convert categories to numbers
data_encoded = pd.get_dummies(data, drop_first=True)
print(f"After encoding: {data_encoded.shape}")

# Step 6: Split features and target
X = data_encoded.drop('HeartDisease', axis=1)
y = data_encoded['HeartDisease']

# Step 7: Train/Test Split (70/30)
X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.3, random_state=42
)

# Step 8: Train Logistic Regression
model = LogisticRegression(max_iter=10000)
model.fit(X_train, y_train)

# Step 9: Make Predictions
y_pred = model.predict(X_test)

# Show first 15 predictions vs actual
print('------Predictions----')
print(y_pred[:15])
print('------Actual-----')
print(y_test[:15].values)

# Step 10: Create results dataframe
data_result = data.loc[X_test.index].copy()
data_result["Predicted"] = y_pred
data_result["Actual"] = y_test.values

# Count errors and correct predictions
errors = (data_result["Predicted"] != data_result["Actual"]).sum()
correct = (data_result["Predicted"] == data_result["Actual"]).sum()
accuracy = correct / len(data_result)

print(f"Errors: {errors}")
print(f"Correct: {correct}")
print(f"Accuracy: {accuracy * 100:.2f}%")

# Step 11: Detailed Evaluation
cm = confusion_matrix(y_test, y_pred)
print('Confusion Matrix:')
print(cm)

# Step 12: Visualize Confusion Matrix
sns.heatmap(cm, annot=True, fmt='d', cmap='Reds')
plt.xlabel('Predicted')
plt.ylabel('Actual')
plt.title('Confusion Matrix Heatmap')
plt.show()

# Extra: Try Decision Tree for comparison
from sklearn.tree import DecisionTreeClassifier

dt_model = DecisionTreeClassifier(random_state=42)
dt_model.fit(X_train, y_train)
y_pred_dt = dt_model.predict(X_test)
acc_dt = accuracy_score(y_test, y_pred_dt)

print(f'\nDecision Tree Accuracy: {acc_dt * 100:.2f}%')
print('Logistic Regression is better!')
